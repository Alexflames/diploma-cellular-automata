\begin{frame}
    \frametitle{Постановка задачи \slnumber}
    \Large
    \begin{enumerate}
      \item создание и оптимизация приложения для моделирования и визуализации клеточных автоматов.
      \item написание конфигурируемого генетического алгоритма для поиска автоматов, воспроизводящих целевое изображение.
      \item сбор данных и их анализ для определения лучших параметров модели.
    \end{enumerate}
\end{frame}
